36  // lb is the corner of AABB with minimal coordinates - left bottom, rt is maximal corner
37  // r.org is origin of ray
38  float t1 = (aabbMin.x - origin.x) * dirfrac.x;

42  float t5 = (aabbMin.z - origin.z) * dirfrac.z;
43  float t6 = (aabbMax.z - origin.z) * dirfrac.z;
44 
45  float tmin = max(max(min(t1, t2), min(t3, t4)), min(t5, t6));
46  float tmax = min(min(max(t1, t2), max(t3, t4)), max(t5, t6));
47 
48  // if tmax < 0, ray (line) is intersecting AABB, but whole AABB is behind us
49  if (tmax < 0)
50  {
51  distance = tmax;
52  return false;
53  }
54
